// Helper to detect compressed fragment buffers

function isCompressedBuffer(buffer: ArrayBuffer): boolean {

    if (buffer.byteLength < 2) return false;

    const intArray = new Uint8Array(buffer);

    if (intArray[0] !== 0x78) return false;

    if (intArray[1] !== 0x01 && intArray[1] !== 0x9C && intArray[1] !== 0xDA) return false;

    return true;

}

src/architecture/ArchitectureFragments.ts
import * as THREE from "three";
import * as OBC from "@thatopen/components";

export class ArchitectureFragments {
  constructor(
    private components: OBC.Components,
    private world: OBC.World
  ) {}

  private fragments = this.components.get(OBC.FragmentsManager);

  async addWall(params: {
    id: string;
    start: THREE.Vector3;
    end: THREE.Vector3;
    height: number;
    thickness: number;
  }) {
    const length = params.start.distanceTo(params.end);

    const geometry = new THREE.BoxGeometry(
      length,
      params.height,
      params.thickness
    );

    geometry.translate(length / 2, params.height / 2, 0);

    const material = new THREE.MeshStandardMaterial({
      color: 0xdcdcdc,
    });

    const mesh = new THREE.Mesh(geometry, material);

    mesh.position.copy(params.start);
    mesh.lookAt(params.end);

    await this.fragments.core.add({
      mesh,
      category: "IFCWALL",
      metadata: {
        id: params.id,
        height: params.height,
        thickness: params.thickness,
      },
    });

    await this.fragments.core.update(true);
  }

  async clear() {
    for (const model of this.fragments.list.values()) {
      model.dispose();
    }
  }
}

import * as THREE from "three";
import { GLTFLoader } from "three/examples/jsm/loaders/GLTFLoader";

export class FurnitureManager {
  private loader = new GLTFLoader();
  private items = new Map<string, THREE.Object3D>();

  constructor(private scene: THREE.Scene) {}

  async add(id: string, url: string, position: THREE.Vector3) {
    const gltf = await this.loader.loadAsync(url);
    const instance = gltf.scene.clone();
    instance.position.copy(position);
    this.scene.add(instance);
    this.items.set(id, instance);
  }

  move(id: string, position: THREE.Vector3) {
    const item = this.items.get(id);
    if (item) item.position.copy(position);
  }
}
